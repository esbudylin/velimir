Collect examples for selected classes (RNC files are required)
#+begin_src python :results table
  import sqlite3

  from bs4 import BeautifulSoup

  from velimir.parsers import extract_lines
  from velimir.io import read_poem_xml
  from velimir.validation import meters_to_str
  from velimir.ml_loader import MeterClassRegistry
  from velimir.settings import PREDICTION_DB_PATH

  MeterClassRegistry.initialize()

  meters = [(i, MeterClassRegistry.int_to_mc(i)) for i in range(MeterClassRegistry.num())]
  examples = []

  conn = sqlite3.connect(PREDICTION_DB_PATH)

  for i, mclass in meters:
      res = conn.execute("""
      SELECT poem_path, line_idx
      FROM predictions
      WHERE meter_class_target = ?
      ORDER BY RANDOM()
      """, (i,))

      poem_path, line_id = res.fetchone()

      poem_xml = read_poem_xml(poem_path)
      soup = BeautifulSoup(poem_xml, "xml")
      lines = list(extract_lines(soup))

      line = lines[line_id]
      caesura_repr = "-".join(map(str, mclass.caesura))

      examples.append((line.text, meters_to_str(mclass), caesura_repr or "-", poem_path, meter_class_counter[mclass]))

  examples
#+end_src
