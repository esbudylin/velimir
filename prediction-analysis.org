Для каждого метра собираем соотношение попаданий при правильно
предсказанном ритме, ошибочно предсказанном ритме и общее соотношение
правильных предсказаний.

#+begin_src sqlite
  SELECT
      meter_target,
      ROUND(AVG(CASE WHEN accent_target == accent_pred THEN meter_target == meter_pred END), 3) AS meter_ok_given_accent_ok,
      ROUND(AVG(CASE WHEN accent_target <> accent_pred THEN meter_target == meter_pred END), 3) AS meter_ok_given_accent_bad,
      ROUND(AVG(accent_pred == accent_target), 3) AS accent_ok_total,
      ROUND(AVG(meter_target == meter_pred), 3) AS meter_ok_total,
      COUNT(*) as total
  FROM predictions
  GROUP BY meter_target
  ORDER BY total DESC;
#+end_src

#+begin_src sqlite
  SELECT
      caesura_target AS caesura,
      meter_target,
      meter_pred,
      COUNT(*) AS cnt
  FROM predictions
  WHERE caesura_target == caesura_pred AND caesura_target <> '--'
  GROUP BY caesura_target, caesura_pred, meter_target, meter_pred
  ORDER BY cnt DESC

Статистика предсказаний стабильности метра

#+begin_src sqlite
  SELECT
      unstable_target,
      unstable_pred,
      COUNT(*) * 1.0 / SUM(COUNT(*)) OVER () AS global_ratio,
      COUNT(*) AS cnt
  FROM predictions
  GROUP BY unstable_target, unstable_pred
  ORDER BY unstable_target, unstable_pred;
#+end_src
